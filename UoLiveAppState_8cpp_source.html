45 PlayerLocation UoLiveAppState::getPlayerLocation()

69 void UoLiveAppState::Init()
70 {
71 #ifdef DEBUG
72  printf("Initializing UO Live App State\n");
73 #endif
74  std::string currentModuleFullPath = Utils::GetCurrentModuleFullPath();
75 
76  uint32_t fileVersionSize = GetFileVersionInfoSizeA(currentModuleFullPath.c_str(), NULL);
77  if (fileVersionSize > 0)
78  {
79  char* pVersionBuffer = new char[fileVersionSize];
80  char* pQueryValueBuffer = NULL;
81 
82  if (GetFileVersionInfo(currentModuleFullPath.c_str(), NULL, fileVersionSize, pVersionBuffer))
83  {
84  uint32_t size = 0;
85  if (VerQueryValue(pVersionBuffer, "\\", reinterpret_cast<VOID FAR* FAR*>(&pQueryValueBuffer), &size))
86  {
87  if (size > 0)
88  {
89  VS_FIXEDFILEINFO *verInfo = (VS_FIXEDFILEINFO *)pQueryValueBuffer;
90  if (verInfo->dwSignature == 0xfeef04bd)
91  {
92  int major = HIWORD(verInfo->dwFileVersionMS);
93  int minor = LOWORD(verInfo->dwFileVersionMS);
94  int buildMajor = HIWORD(verInfo->dwFileVersionLS);
95  int buildMinor = LOWORD(verInfo->dwFileVersionLS);
96 
97  char buffer[20];
98  memset(buffer, 0x00, 20);
99  sprintf(buffer, "%i.%i.%i.%i", major, minor, buildMajor, buildMinor);
100 
101  m_versionMajor = verInfo->dwFileVersionMS;
102  m_versionMinor = verInfo->dwFileVersionLS;
103  }
104  }
105  }
106  }
107 
108  delete pVersionBuffer;
109  }
110 
111  m_pFileManager = FileManagerFactory::CreateFileManager(m_versionMajor, m_versionMinor);
112  m_pFileManager->Initialize();
113 
114  m_pNetworkManager = new NetworkManager(this);
115  m_pNetworkManager->init(m_versionMajor, m_versionMinor);
116 
117  m_pLoginHandler = new LoginHandler(m_pNetworkManager);
118  m_pLoginHandler->init();
119 
120  m_pAtlas = new Atlas(m_pFileManager, this, m_pNetworkManager);
121  m_pAtlas->init();
122 
123  m_pMapDimensions = MasterControlUtils::GetMapDimensionAddress();
124 
125 #ifdef DEBUG
126  printf("MAP DIMENSION POINTER: 0x%x\n", m_pMapDimensions);
127 #endif
128 }
